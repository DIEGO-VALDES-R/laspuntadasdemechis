import React, { useState, useEffect } from 'react';
import { useNavigate } from 'react-router-dom';
import jsPDF from 'jspdf';
import autoTable from 'jspdf-autotable';
import { db } from '../services/db';
import { InventoryItem, Client } from '../types';
import { 
  FileText, Download, Send, ChevronLeft, 
  DollarSign, Package, Image as ImageIcon, User 
} from 'lucide-react';

const COLORS = {
  coral: '#D4735E',
  sageGreen: '#A9B4A1',
};

interface QuoteData {
  clientName: string;
  clientEmail: string;
  clientPhone: string;
  productName: string;
  description: string;
  basePrice: number;
  selectedSize: InventoryItem | null;
  selectedPackaging: InventoryItem | null;
  selectedAccessories: InventoryItem[];
  imageUrl: string;
  notes: string;
}

const QuoteGenerator: React.FC = () => {
  const navigate = useNavigate();
  const [loading, setLoading] = useState(true);
  const [clients, setClients] = useState<Client[]>([]);
  const [inventoryItems, setInventoryItems] = useState<InventoryItem[]>([]);
  
  const [quoteData, setQuoteData] = useState<QuoteData>({
    clientName: '',
    clientEmail: '',
    clientPhone: '',
    productName: '',
    description: '',
    basePrice: 0,
    selectedSize: null,
    selectedPackaging: null,
    selectedAccessories: [],
    imageUrl: '',
    notes: ''
  });

  useEffect(() => {
    loadData();
  }, []);

  const loadData = async () => {
    setLoading(true);
    try {
      const [clientsData, inventoryData] = await Promise.all([
        db.getAllClients(),
        db.getInventoryItems()
      ]);
      setClients(clientsData);
      setInventoryItems(inventoryData);
    } catch (error) {
      console.error('Error loading data:', error);
    } finally {
      setLoading(false);
    }
  };

  const handleClientSelect = (email: string) => {
    const client = clients.find(c => c.email === email);
    if (client) {
      setQuoteData({
        ...quoteData,
        clientName: client.nombre_completo,
        clientEmail: client.email,
        clientPhone: client.telefono || ''
      });
    }
  };

  const handleImageUpload = (e: React.ChangeEvent<HTMLInputElement>) => {
    const file = e.target.files?.[0];
    if (file) {
      const reader = new FileReader();
      reader.onload = (ev) => {
        setQuoteData({ ...quoteData, imageUrl: ev.target?.result as string });
      };
      reader.readAsDataURL(file);
    }
  };

  const calculateTotal = () => {
    let total = quoteData.basePrice;
    if (quoteData.selectedSize) total += quoteData.selectedSize.price;
    if (quoteData.selectedPackaging) total += quoteData.selectedPackaging.price;
    quoteData.selectedAccessories.forEach(acc => total += acc.price);
    return total;
  };

  const generatePDF = () => {
    const doc = new jsPDF();
    const total = calculateTotal();

    // Colores de marca
    const coralRGB = [212, 115, 94];
    const sageGreenRGB = [169, 180, 161];

    // Header con logo y t√≠tulo
    doc.setFillColor(...coralRGB);
    doc.rect(0, 0, 210, 40, 'F');
    
    doc.setTextColor(255, 255, 255);
    doc.setFontSize(24);
    doc.setFont('helvetica', 'bold');
    doc.text('PUNTADAS DE MECHIS', 105, 20, { align: 'center' });
    
    doc.setFontSize(12);
    doc.setFont('helvetica', 'normal');
    doc.text('Cotizaci√≥n de Amigurumi', 105, 30, { align: 'center' });

    // Informaci√≥n del cliente
    doc.setTextColor(0, 0, 0);
    doc.setFontSize(11);
    doc.setFont('helvetica', 'bold');
    doc.text('INFORMACI√ìN DEL CLIENTE', 20, 55);
    
    doc.setFont('helvetica', 'normal');
    doc.setFontSize(10);
    doc.text(`Nombre: ${quoteData.clientName}`, 20, 65);
    doc.text(`Email: ${quoteData.clientEmail}`, 20, 72);
    doc.text(`Tel√©fono: ${quoteData.clientPhone}`, 20, 79);
    doc.text(`Fecha: ${new Date().toLocaleDateString('es-CO')}`, 20, 86);

    // Informaci√≥n del producto
    doc.setFont('helvetica', 'bold');
    doc.setFontSize(11);
    doc.text('DETALLES DEL PRODUCTO', 20, 100);
    
    doc.setFont('helvetica', 'normal');
    doc.setFontSize(10);
    doc.text(`Producto: ${quoteData.productName}`, 20, 110);
    
    // Descripci√≥n con salto de l√≠nea
    if (quoteData.description) {
      const splitDescription = doc.splitTextToSize(quoteData.description, 170);
      doc.text(`Descripci√≥n: ${splitDescription}`, 20, 117);
    }

    // Imagen del producto (si existe)
    let yPosition = 135;
    if (quoteData.imageUrl) {
      try {
        doc.addImage(quoteData.imageUrl, 'JPEG', 20, yPosition, 60, 60);
        yPosition += 70;
      } catch (error) {
        console.error('Error adding image:', error);
      }
    }

    // Tabla de desglose de precios
    const tableData: any[] = [];
    
    tableData.push(['Precio Base del Amigurumi', `$${quoteData.basePrice.toLocaleString()}`]);
    
    if (quoteData.selectedSize) {
      tableData.push([`Tama√±o: ${quoteData.selectedSize.label}`, `$${quoteData.selectedSize.price.toLocaleString()}`]);
    }
    
    if (quoteData.selectedPackaging) {
      tableData.push([`Empaque: ${quoteData.selectedPackaging.label}`, `$${quoteData.selectedPackaging.price.toLocaleString()}`]);
    }
    
    quoteData.selectedAccessories.forEach(acc => {
      tableData.push([`Accesorio: ${acc.label}`, `$${acc.price.toLocaleString()}`]);
    });

    autoTable(doc, {
      startY: yPosition,
      head: [['Concepto', 'Valor']],
      body: tableData,
      foot: [['TOTAL', `$${total.toLocaleString()}`]],
      theme: 'striped',
      headStyles: { fillColor: coralRGB, textColor: [255, 255, 255], fontSize: 11, fontStyle: 'bold' },
      footStyles: { fillColor: sageGreenRGB, textColor: [255, 255, 255], fontSize: 12, fontStyle: 'bold' },
      styles: { fontSize: 10 },
      margin: { left: 20, right: 20 }
    });

    // Notas adicionales
    if (quoteData.notes) {
      const finalY = (doc as any).lastAutoTable.finalY + 15;
      doc.setFont('helvetica', 'bold');
      doc.setFontSize(11);
      doc.text('NOTAS ADICIONALES:', 20, finalY);
      
      doc.setFont('helvetica', 'normal');
      doc.setFontSize(9);
      const splitNotes = doc.splitTextToSize(quoteData.notes, 170);
      doc.text(splitNotes, 20, finalY + 7);
    }

    // Footer
    doc.setFillColor(...sageGreenRGB);
    doc.rect(0, 280, 210, 17, 'F');
    
    doc.setTextColor(255, 255, 255);
    doc.setFontSize(9);
    doc.text('Puntadas de Mechis | WhatsApp: +57 312 491 5127 | Email: info@puntadasdemechis.com', 105, 290, { align: 'center' });

    // Guardar PDF
    const filename = `Cotizacion_${quoteData.clientName.replace(/\s/g, '_')}_${Date.now()}.pdf`;
    doc.save(filename);
  };

  const sendQuoteByEmail = () => {
    const total = calculateTotal();
    const subject = `Cotizaci√≥n - ${quoteData.productName}`;
    const body = `
Hola ${quoteData.clientName},

Te enviamos la cotizaci√≥n para tu amigurumi personalizado:

üì¶ Producto: ${quoteData.productName}
üìù Descripci√≥n: ${quoteData.description}

üí∞ Desglose de Precios:
- Precio Base: $${quoteData.basePrice.toLocaleString()}
${quoteData.selectedSize ? `- Tama√±o (${quoteData.selectedSize.label}): $${quoteData.selectedSize.price.toLocaleString()}` : ''}
${quoteData.selectedPackaging ? `- Empaque (${quoteData.selectedPackaging.label}): $${quoteData.selectedPackaging.price.toLocaleString()}` : ''}
${quoteData.selectedAccessories.map(acc => `- Accesorio (${acc.label}): $${acc.price.toLocaleString()}`).join('\n')}

üíµ TOTAL: $${total.toLocaleString()}

${quoteData.notes ? `\nüìå Notas: ${quoteData.notes}` : ''}

¬°Gracias por tu inter√©s!

Puntadas de Mechis
    `.trim();

    const mailtoLink = `mailto:${quoteData.clientEmail}?subject=${encodeURIComponent(subject)}&body=${encodeURIComponent(body)}`;
    window.location.href = mailtoLink;
  };

  if (loading) {
    return (
      <div className="flex items-center justify-center min-h-screen">
        <div className="animate-spin rounded-full h-12 w-12 border-b-2 border-pink-600"></div>
      </div>
    );
  }

  const sizes = inventoryItems.filter(item => item.category === 'sizes');
  const packaging = inventoryItems.filter(item => item.category === 'packaging');
  const accessories = inventoryItems.filter(item => item.category === 'accessories');

  return (
    <div className="min-h-screen bg-gradient-to-br from-pink-50 via-purple-50 to-orange-50 p-8">
      <div className="max-w-5xl mx-auto">
        {/* Header */}
        <div className="mb-8">
          <button
            onClick={() => navigate('/admin')}
            className="flex items-center gap-2 text-gray-600 hover:text-gray-800 mb-4"
          >
            <ChevronLeft size={20} />
            Volver al Panel
          </button>
          
          <div className="flex items-center gap-3 mb-2">
            <div className="p-3 bg-gradient-to-r from-pink-500 to-purple-600 rounded-xl">
              <FileText className="text-white" size={32} />
            </div>
            <div>
              <h1 className="text-3xl font-bold text-gray-800">Generador de Cotizaciones</h1>
              <p className="text-gray-600">Crea cotizaciones profesionales en PDF para tus clientes</p>
            </div>
          </div>
        </div>

        {/* Formulario */}
        <div className="bg-white rounded-2xl shadow-lg border border-gray-100 p-8 space-y-6">
          
          {/* Informaci√≥n del Cliente */}
          <section className="border-b pb-6">
            <h2 className="text-xl font-bold text-gray-800 mb-4 flex items-center gap-2">
              <User size={20} style={{ color: COLORS.coral }} />
              Informaci√≥n del Cliente
            </h2>
            
            <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
              <div>
                <label className="block text-sm font-medium text-gray-700 mb-2">
                  Seleccionar Cliente Existente
                </label>
                <select
                  onChange={(e) => handleClientSelect(e.target.value)}
                  className="w-full px-4 py-2 border border-gray-300 rounded-lg focus:ring-2 focus:ring-pink-500 focus:border-transparent"
                >
                  <option value="">-- Nuevo Cliente --</option>
                  {clients.map(client => (
                    <option key={client.id} value={client.email}>
                      {client.nombre_completo} ({client.email})
                    </option>
                  ))}
                </select>
              </div>
            </div>

            <div className="grid grid-cols-1 md:grid-cols-3 gap-4 mt-4">
              <div>
                <label className="block text-sm font-medium text-gray-700 mb-2">Nombre *</label>
                <input
                  type="text"
                  value={quoteData.clientName}
                  onChange={(e) => setQuoteData({ ...quoteData, clientName: e.target.value })}
                  className="w-full px-4 py-2 border border-gray-300 rounded-lg focus:ring-2 focus:ring-pink-500 focus:border-transparent"
                  placeholder="Juan P√©rez"
                />
              </div>
              
              <div>
                <label className="block text-sm font-medium text-gray-700 mb-2">Email *</label>
                <input
                  type="email"
                  value={quoteData.clientEmail}
                  onChange={(e) => setQuoteData({ ...quoteData, clientEmail: e.target.value })}
                  className="w-full px-4 py-2 border border-gray-300 rounded-lg focus:ring-2 focus:ring-pink-500 focus:border-transparent"
                  placeholder="cliente@email.com"
                />
              </div>
              
              <div>
                <label className="block text-sm font-medium text-gray-700 mb-2">Tel√©fono</label>
                <input
                  type="tel"
                  value={quoteData.clientPhone}
                  onChange={(e) => setQuoteData({ ...quoteData, clientPhone: e.target.value })}
                  className="w-full px-4 py-2 border border-gray-300 rounded-lg focus:ring-2 focus:ring-pink-500 focus:border-transparent"
                  placeholder="3001234567"
                />
              </div>
            </div>
          </section>

          {/* Detalles del Producto */}
          <section className="border-b pb-6">
            <h2 className="text-xl font-bold text-gray-800 mb-4 flex items-center gap-2">
              <Package size={20} style={{ color: COLORS.sageGreen }} />
              Detalles del Producto
            </h2>
            
            <div className="space-y-4">
              <div>
                <label className="block text-sm font-medium text-gray-700 mb-2">Nombre del Producto *</label>
                <input
                  type="text"
                  value={quoteData.productName}
                  onChange={(e) => setQuoteData({ ...quoteData, productName: e.target.value })}
                  className="w-full px-4 py-2 border border-gray-300 rounded-lg focus:ring-2 focus:ring-pink-500 focus:border-transparent"
                  placeholder="Ej: Osito Teddy Personalizado"
                />
              </div>
              
              <div>
                <label className="block text-sm font-medium text-gray-700 mb-2">Descripci√≥n</label>
                <textarea
                  value={quoteData.description}
                  onChange={(e) => setQuoteData({ ...quoteData, description: e.target.value })}
                  className="w-full px-4 py-2 border border-gray-300 rounded-lg focus:ring-2 focus:ring-pink-500 focus:border-transparent"
                  rows={3}
                  placeholder="Detalles del amigurumi..."
                />
              </div>

              <div>
                <label className="block text-sm font-medium text-gray-700 mb-2">Precio Base * ($)</label>
                <input
                  type="number"
                  value={quoteData.basePrice}
                  onChange={(e) => setQuoteData({ ...quoteData, basePrice: Number(e.target.value) })}
                  className="w-full px-4 py-2 border border-gray-300 rounded-lg focus:ring-2 focus:ring-pink-500 focus:border-transparent"
                  placeholder="50000"
                  min="0"
                />
              </div>

              <div>
                <label className="block text-sm font-medium text-gray-700 mb-2">Imagen del Producto</label>
                {quoteData.imageUrl && (
                  <img src={quoteData.imageUrl} alt="Preview" className="w-40 h-40 object-cover rounded-lg mb-2 border-2 border-gray-200" />
                )}
                <input
                  type="file"
                  accept="image/*"
                  onChange={handleImageUpload}
                  className="text-sm"
                />
              </div>
            </div>
          </section>

          {/* Opciones Adicionales */}
          <section className="border-b pb-6">
            <h2 className="text-xl font-bold text-gray-800 mb-4 flex items-center gap-2">
              <DollarSign size={20} style={{ color: COLORS.coral }} />
              Opciones Adicionales
            </h2>
            
            <div className="grid grid-cols-1 md:grid-cols-3 gap-4">
              {/* Tama√±o */}
              <div>
                <label className="block text-sm font-medium text-gray-700 mb-2">üìè Tama√±o</label>
                <select
                  onChange={(e) => {
                    const item = sizes.find(s => s.id === e.target.value);
                    setQuoteData({ ...quoteData, selectedSize: item || null });
                  }}
                  className="w-full px-4 py-2 border border-gray-300 rounded-lg focus:ring-2 focus:ring-pink-500 focus:border-transparent"
                >
                  <option value="">-- Ninguno --</option>
                  {sizes.map(size => (
                    <option key={size.id} value={size.id}>
                      {size.label} (+${size.price.toLocaleString()})
                    </option>
                  ))}
                </select>
              </div>

              {/* Empaque */}
              <div>
                <label className="block text-sm font-medium text-gray-700 mb-2">üì¶ Empaque</label>
                <select
                  onChange={(e) => {
                    const item = packaging.find(p => p.id === e.target.value);
                    setQuoteData({ ...quoteData, selectedPackaging: item || null });
                  }}
                  className="w-full px-4 py-2 border border-gray-300 rounded-lg focus:ring-2 focus:ring-pink-500 focus:border-transparent"
                >
                  <option value="">-- Ninguno --</option>
                  {packaging.map(pack => (
                    <option key={pack.id} value={pack.id}>
                      {pack.label} (+${pack.price.toLocaleString()})
                    </option>
                  ))}
                </select>
              </div>

              {/* Accesorios */}
              <div>
                <label className="block text-sm font-medium text-gray-700 mb-2">‚ú® Accesorios</label>
                <div className="space-y-2 max-h-40 overflow-y-auto border border-gray-200 rounded-lg p-3">
                  {accessories.map(acc => (
                    <label key={acc.id} className="flex items-center gap-2 cursor-pointer">
                      <input
                        type="checkbox"
                        onChange={(e) => {
                          if (e.target.checked) {
                            setQuoteData({
                              ...quoteData,
                              selectedAccessories: [...quoteData.selectedAccessories, acc]
                            });
                          } else {
                            setQuoteData({
                              ...quoteData,
                              selectedAccessories: quoteData.selectedAccessories.filter(a => a.id !== acc.id)
                            });
                          }
                        }}
                        className="rounded text-pink-600 focus:ring-pink-500"
                      />
                      <span className="text-sm">{acc.label} (+${acc.price.toLocaleString()})</span>
                    </label>
                  ))}
                </div>
              </div>
            </div>
          </section>

          {/* Notas */}
          <section className="border-b pb-6">
            <h2 className="text-xl font-bold text-gray-800 mb-4">üìå Notas Adicionales</h2>
            <textarea
              value={quoteData.notes}
              onChange={(e) => setQuoteData({ ...quoteData, notes: e.target.value })}
              className="w-full px-4 py-2 border border-gray-300 rounded-lg focus:ring-2 focus:ring-pink-500 focus:border-transparent"
              rows={3}
              placeholder="T√©rminos, condiciones, tiempo de entrega, etc."
            />
          </section>

          {/* Total y Acciones */}
          <section>
            <div className="bg-gradient-to-r from-pink-100 to-purple-100 rounded-xl p-6 mb-6">
              <div className="flex justify-between items-center">
                <span className="text-2xl font-bold text-gray-800">TOTAL:</span>
                <span className="text-4xl font-bold text-transparent bg-clip-text bg-gradient-to-r from-pink-600 to-purple-600">
                  ${calculateTotal().toLocaleString()}
                </span>
              </div>
            </div>

            <div className="flex gap-4">
              <button
                onClick={generatePDF}
                disabled={!quoteData.clientName || !quoteData.clientEmail || !quoteData.productName || quoteData.basePrice === 0}
                className="flex-1 bg-gradient-to-r from-pink-600 to-purple-600 text-white px-8 py-4 rounded-xl font-bold hover:from-pink-700 hover:to-purple-700 transition-all shadow-lg disabled:opacity-50 disabled:cursor-not-allowed flex items-center justify-center gap-2"
              >
                <Download size={20} />
                Generar y Descargar PDF
              </button>

              <button
                onClick={sendQuoteByEmail}
                disabled={!quoteData.clientEmail}
                className="flex-1 bg-gradient-to-r from-green-600 to-teal-600 text-white px-8 py-4 rounded-xl font-bold hover:from-green-700 hover:to-teal-700 transition-all shadow-lg disabled:opacity-50 disabled:cursor-not-allowed flex items-center justify-center gap-2"
              >
                <Send size={20} />
                Enviar por Email
              </button>
            </div>
          </section>
        </div>
      </div>
    </div>
  );
};

export default QuoteGenerator;
